## RichTooltip

<ComponentDetail of="RichTooltip" github="Tooltip/RichTooltip" />

<Playground>
  {() => {
    const [color, setColor] = React.useState({
      h: 140,
      s: 1,
      v: 1,
    })
    function handleColorStateChange(hs) {
      setColor({
        ...hs,
        v: color.v,
      })
    }
    const popoverContent = (
      <RichTooltipContent>
        <ColorWheel
          hue={color.h}
          saturation={color.s}
          value={color.v}
          onColorChange={handleColorStateChange}
        />
      </RichTooltipContent>
    )
    const colorHsv = hsv(color.h, color.s, color.v)
    return (
      <Box display="inline-block" mr="small">
        <RichTooltip placement="top" content={popoverContent}>
          {(eventHandlers, ref) => (
            <Box
              ref={ref}
              bg={colorHsv.toString()}
              p="medium"
              borderRadius="4px"
              {...eventHandlers}
            >
              Pick a new color.
            </Box>
          )}
        </RichTooltip>
      </Box>
    )
  }}
</Playground>
